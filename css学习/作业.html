账单计算伪代码

// 读取订单
function readOrder(order) {
    return order.items;
}

// 计算总价（并行过程1）
function calculateTotalPrice(items) {
    let total = 0;
    for (let item of items) {
        total += item.price * item.quantity;
    }
    return total;
}

// 确定折扣率（并行过程2）
function determineDiscountRate(order) {
    if (order.customer.isVIP) {
        return 0.1; // VIP客户享受10%折扣
    } else if (order.totalAmount > 200) {
        return 0.05; // 大额订单享受5%折扣
    }
    return 0; // 无折扣
}

// 确定服务费率（并行过程3）
function determineServiceChargeRate(order) {
    if (order.dineIn) {
        return 0.1; // 堂食收取10%服务费
    }
    return 0; // 外卖无服务费
}

// 应用折扣计算折扣后价格
function applyDiscount(totalPrice, discountRate) {
    return totalPrice * (1 - discountRate);
}

// 应用服务费计算净价
function applyServiceCharge(discountedPrice, serviceChargeRate) {
    return discountedPrice * (1 + serviceChargeRate);
}

// 主流程
function calculateBill(order) {
    const items = readOrder(order);
    
    // 并行计算
    const totalPrice = calculateTotalPrice(items);
    const discountRate = determineDiscountRate(order);
    const serviceChargeRate = determineServiceChargeRate(order);
    
    // 应用折扣和服务费
    const discountedPrice = applyDiscount(totalPrice, discountRate);
    const netPrice = applyServiceCharge(discountedPrice, serviceChargeRate);
    
    return netPrice;
}

活动图说明

  • 泳道：使用泳道清晰区分服务员和系统的活动范围
  • 并行处理：总价计算、折扣确定和服务费确定三个过程并行执行，提高效率
  • 分叉与合并节点：使用分叉节点开始并行处理，使用合并节点同步所有并行活动
  • 复杂过程分离：按照要求，三个计算过程分离为独立活动，不合并
  • 顺序依赖：折扣应用必须在总价和折扣率计算完成后进行，服务费应用必须在折扣后价格和服务费率计算完成后进行
